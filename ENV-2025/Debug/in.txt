integer function string_compare(char str1, char str2)
    is str1 == str2
    true [
        return 1
    ]
    false [
        return 0
    ]
end

integer function calculate_power(integer base, integer exponent)
    type integer result = 1
    type integer i = 1
    
    repeat exponent times [
        result = result * base
    ]
    
    return result
end

main
{
    write "=== ДЕМОНСТРАЦИЯ ЦЕЛОЧИСЛЕННЫХ ОПЕРАЦИЙ ==="
    writeline ""
    
    type integer a = 0b1101
    type integer b = 0x5
    
    write "a (двоичное 0b1101) = "
    writeline a
    write "b (шестнадцатеричное 0x5) = "
    writeline b
    
    write "a + b = "
    writeline a + b
    write "a - b = "
    writeline a - b
    write "a * b = "
    writeline a * b
    write "a / b = "
    writeline a / b
    write "a % b = "
    writeline a % b
    
    writeline ""
    write "=== ДЕМОНСТРАЦИЯ ЦИКЛА ==="
    writeline ""
    
    type integer counter = 1
    repeat 4 times [
        write "Итерация "
        write counter
        write ": значение = "
        writeline counter * 2
        counter = counter + 1
    ]
    
    writeline ""
    write "=== ДЕМОНСТРАЦИЯ СИМВОЛЬНЫХ ОПЕРАЦИЙ ==="
    writeline ""
    
    type char char1 = 'A'
    type char char2 = 'B'
    type char char3 = 'A'
    
    write "Символ 1: "
    writeline char1
    write "Символ 2: "
    writeline char2
    write "Символ 3: "
    writeline char3
    
    type integer compare_result1 = string_compare(char1, char2)
    write "Сравнение A и B: "
    writeline compare_result1
    
    type integer compare_result2 = string_compare(char1, char3)
    write "Сравнение A и A: "
    writeline compare_result2
    
    writeline ""
    write "=== ДЕМОНСТРАЦИЯ ПРОИЗВОЛЬНОЙ ФУНКЦИИ ==="
    writeline ""
    
    type integer power_result = calculate_power(2, 4)
    write "2 в степени 4 = "
    writeline power_result
    
    writeline ""
    write "=== ДЕМОНСТРАЦИЯ ЛОГИЧЕСКИХ ЛИТЕРАЛОВ ==="
    writeline ""
    
    write "Логические значения: "
    write true
    write " "
    write false
    writeline ""
    
    return 0
}
