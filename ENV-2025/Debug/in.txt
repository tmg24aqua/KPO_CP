integer function add(integer a, integer b)
    return a + b
end

integer function subtract(integer x, integer y)
    return x - y
end

integer function multiply(integer num1, integer num2)
    return num1 * num2
end

integer function divide(integer dividend, integer divisor)
    is divisor & 0
    true [
        return 0
    ]
    false [
        return dividend / divisor
    ]
end

integer function modulo(integer value, integer mod)
    return value % mod
end

integer function max(integer a, integer b)
    is a > b
    true [
        return a
    ]
    false [
        return b
    ]
end

integer function min(integer first, integer second)
    is first < second
    true [
        return first
    ]
    false [
        return second
    ]
end

integer function power(integer base, integer exponent)
    type integer result = 1
    type integer i = 1
    repeat i times exponent [
        result = result * base
        i = i + 1
    ]
    return result
end

integer function isEven(integer number)
    is number % 2 & 0
    true [
        return 1
    ]
    false [
        return 0
    ]
end

integer function factorial(integer n)
    type integer result = 1
    type integer counter = 1
    repeat counter times n [
        result = result * counter
        counter = counter + 1
    ]
    return result
end

integer function cmpStrs(char str1, char str2)
    return compare(str1, str2)
end

integer function getStrLen(symbol text)
    return length(text)
end

integer function calculate(integer x, integer y, integer z)
    type integer sum = x + y + z
    type integer product = x * y * z
    type integer average = sum / 3
    return average
end

integer function strOps(symbol str1, symbol str2)
    type integer len1 = length(str1)
    type integer len2 = length(str2)
    type integer totalLen = len1 + len2
    return totalLen
end

integer function chkEqu(char str1, char str2)
    return compare(str1, str2)
end
